```bash
# init
sudo apt install python3-tk

```

```bash
# Chainer 4.0.0
sudo pip3 install chainer==4.0.0

wget https://github.com/chainer/chainer/archive/v4.0.0.tar.gz
tar xzf v4.0.0.tar.gz
python3 chainer-4.0.0/examples/mnist/train_mnist.py

# Chainer 7.8.1
sudo pip3 install chainer==7.8.1

wget https://github.com/chainer/chainer/archive/v7.8.1.tar.gz
tar xzf v7.8.1.tar.gz
python3 chainer-7.8.1/examples/mnist/train_mnist.py

```
